
CREATE TABLE "Users" (
    id SERIAL PRIMARY KEY,
    login VARCHAR(50) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    role VARCHAR(20) DEFAULT 'User',
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE "Authors" (
    id SERIAL PRIMARY KEY,
    first_name VARCHAR(100) NOT NULL,
    last_name VARCHAR(100) NOT NULL,
    birth_date DATE,
    biography TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE "Genres" (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE "Books" (
    id SERIAL PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    author_id INTEGER NOT NULL REFERENCES "Authors"(id) ON DELETE CASCADE,
    genre_id INTEGER NOT NULL REFERENCES "Genres"(id) ON DELETE CASCADE,
    year_published INTEGER CHECK (year_published >= 1000 AND year_published <= EXTRACT(YEAR FROM CURRENT_DATE)),
    isbn VARCHAR(20) UNIQUE,
    description TEXT,
    page_count INTEGER,
    publisher VARCHAR(200),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_books_title ON "Books" (title);
CREATE INDEX idx_books_author ON "Books" (author_id);
CREATE INDEX idx_books_genre ON "Books" (genre_id);
CREATE INDEX idx_books_year ON "Books" (year_published);
CREATE INDEX idx_authors_name ON "Authors" (last_name, first_name);
CREATE INDEX idx_genres_name ON "Genres" (name);
CREATE INDEX idx_users_login ON "Users" (login);

CREATE OR REPLACE FUNCTION "RegisterUser"(
    p_login VARCHAR,
    p_password_hash VARCHAR
) RETURNS INTEGER
LANGUAGE plpgsql
AS $$
BEGIN
    IF EXISTS (SELECT 1 FROM "Users" WHERE login = p_login) THEN
        RETURN 0;
    END IF;
    INSERT INTO "Users" (login, password_hash, role, is_active)
    VALUES (p_login, p_password_hash, 'User', true);
    RETURN 1;
END;
$$;

CREATE OR REPLACE FUNCTION "SearchBooks"(
    p_title VARCHAR DEFAULT NULL,
    p_author VARCHAR DEFAULT NULL,
    p_genre VARCHAR DEFAULT NULL,
    p_year_from INTEGER DEFAULT NULL,
    p_year_to INTEGER DEFAULT NULL
)
RETURNS TABLE(
    book_id INTEGER,
    title VARCHAR,
    author_name TEXT,
    genre_name VARCHAR,
    year_published INTEGER,
    isbn VARCHAR,
    page_count INTEGER,
    publisher VARCHAR
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    SELECT
        b.id,
        b.title,
        CONCAT(a.last_name, ' ', a.first_name),
        g.name,
        b.year_published,
        b.isbn,
        b.page_count,
        b.publisher
    FROM "Books" b
    JOIN "Authors" a ON b.author_id = a.id
    JOIN "Genres" g ON b.genre_id = g.id
    WHERE (p_title IS NULL OR b.title ILIKE '%' || p_title || '%')
      AND (p_author IS NULL OR CONCAT(a.last_name, ' ', a.first_name) ILIKE '%' || p_author || '%')
      AND (p_genre IS NULL OR g.name ILIKE '%' || p_genre || '%')
      AND (p_year_from IS NULL OR b.year_published >= p_year_from)
      AND (p_year_to IS NULL OR b.year_published <= p_year_to)
    ORDER BY b.title;
END;
$$;

    isbn VARCHAR,
    description TEXT,
    page_count INTEGER,
    publisher VARCHAR,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    SELECT
        b.id,
        b.title,
        b.author_id,
        CONCAT(a.last_name, ' ', a.first_name),
        b.genre_id,
        g.name,
        b.year_published,
        b.isbn,
        b.description,
        b.page_count,
        b.publisher,
        b.created_at,
        b.updated_at
    FROM "Books" b

$$;

CREATE OR REPLACE FUNCTION "UpdateUpdatedAtColumn"()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_update_books_updated_at
    BEFORE UPDATE ON "Books"
    FOR EACH ROW
    EXECUTE FUNCTION "UpdateUpdatedAtColumn"();
